\begin{python}
import numpy as np
import pandas as pd
from scipy.optimize import leastsq
import matplotlib.pyplot as plt


class PolyLeastSquaresFitting:
    """Applies a polynomial least squares fitting to data, by default,
    the power index of polynomial is 3.

    Tries to make the choice of basis function and weight more easier

    Attributes:
        __init__: A function initializing the class object
        read_data: A function loading data to be fitted
        fit: A function applying specific fit to data
        plot_result: A function displaying the multiply fitting results
    """

    def __init__(self, generate_flag=True, power_index=3):
        """Initializes the class object by parameter assignment"""
        self.generate_flag = generate_flag
        self.data_path = None
        self.X = None
        self.y = None
        self.y_fitted = None
        self.power_index = power_index
        self.res = None

    def read_data(self):
        """Loads prepared data to fit"""
        if self.generate_flag:
            self.X = np.linspace(-1, 2, 7)
            self.y = np.array([
                -4.447, -0.452, 0.551, 0.048, -0.447, 0.549, 4.552
            ])
        else:
            print("Please specify th file path of data: ")
            self.data_path = input()
            data = pd.read_txt(self.data_path)
            try:
                self.X = data["X"]
                self.y = data["y"]
            except IndexError:
                print(
                    "Please check the formation of data file." +
                    "Make sure the column name of independent variable" +
                    " and dependent variable is named as X, y respectively"
                )


    def fit(self):
        """Applies polynomial least squares fitting"""
        def func(x, params):
            """Specifies the formation of fit function"""
            a1, a2, a3, a4 = params
            return a1 + a2 * x + a3 * x ** 2 + a4 * x ** 3

        def residuals(params, y, x):
            """Specifies the formation of residuals between fitted value and real value"""
            return y - func(x, params)

        params0 = np.array([1, 1, 1, 1])
        self.res = leastsq(residuals, params0, args=(self.y, self.X))
        self.y_fitted = func(self.X, self.res[0])

    def plot_result(self):
        """Plots specific results"""
        def plot_params():
            """Plots fitted parameters"""
            print("The fitted parameters is listed below: ")
            for ind, param in enumerate(self.res[0]):
                print("The fitted value of a" + "{} is: {}".format(ind, param))

        def plot_se():
            """Plots square error"""
            se = ((self.y - self.y_fitted) ** 2).sum()
            print("The square error of fitting is {}".format(se))

        def plot_fittedCurve():
            """Plots fitted curve"""
            if self.power_index == 3:
                plt.plot(self.X, self.y, "ro", label="Real Data");
                plt.plot(self.X, self.y_fitted, "b-", label="Fitted curve by poly least squares with 3 power");
                plt.legend();
                plt.show();
            else:
                pass

        plot_params()
        plot_se()
        plot_fittedCurve()


case = PolyLeastSquaresFitting()
case.read_data()
case.fit()
case.plot_result()
\end{python}

\begin{python}
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt


class RungeKutta:
    """Applies a Runge-Kutta to solve differential equation group, by default,
    the oder of is Runge-Kutta 4.

    Attributes:
        __init__: A function initializing the class object
        fit: A function solving specific equation
        plot_result: A function displaying the multiply fitting results
    """

    def __init__(self, x0=0, y0=1, params=-50, order=4, step=0.01):
        """Initializes the class object by parameter assignment"""
        self.x0 = x0
        self.y0 = y0
        self.params = params
        self.x_list = []
        self.y_sol = []
        self.y_exact = None
        self.order = order
        self.step = step
        self.res = None

    def fit(self):
        """Applies polynomial least squares fitting"""
        def load_equ(y, x):
            """Loads equation to solve"""
            a = self.params
            return a * y - a * x + 1

        def runge_kutta_4(y, x, dx, func):
            """ Applies 4 oder runge-kutta algorithm"""
            k1 = dx * func(y, x)
            k2 = dx * func(y + 0.5 * k1, x + 0.5 * dx)
            k3 = dx * func(y + 0.5 * k2, x + 0.5 * dx)
            k4 = dx * func(y + k3, x + dx)
            return y + (k1 + 2 * k2 + 2 * k3 + k4) / 6.

        if self.order == 4:
            x = self.x0
            y = self.y0
            while x < 1:
                y = runge_kutta_4(y, x, self.step, load_equ)
                x += self.step
                self.x_list.append(x)
                self.y_sol.append(y)

    def plot_result(self):
        """Plots specific results"""
        self.y_exact = [np.exp(self.params * x_i) + x_i for x_i in self.x_list]

        def plot_se():
            """Plots square error"""
            se = ((np.array(self.y_exact) - np.array(self.y_sol)) ** 2)
            plt.figure(num=1)
            plt.plot(self.x_list, se, "k-", label="Square Error");
            plt.legend();
            plt.show();

        def plot_solvedValue():
            """Plots simulated value for each point"""
            if self.order == 4:
                table = pd.DataFrame(
                    {
                        "Selected_point": self.x_list,
                        "Exact_Data_Point": self.y_exact,
                        "Solve_Data_Point": self.y_sol
                    }
                )
                print("The comparison table between exact value and " +
                    "solved data point is printed here: ")
                print(table)
            else:
                pass

        def plot_solvedCurve():
            """Plots solved curve"""
            if self.order == 4:
                plt.figure(num=2)
                plt.plot(
                    self.x_list,
                    np.log10(np.array(self.y_exact)),
                    "ro",
                    MarkerFaceColor="w",
                    MarkerEdgeColor="r",
                    MarkerSize=5,
                    label="Exact Solution, a=-50"
                );
                plt.plot(
                    self.x_list,
                    np.log10(np.array(self.y_sol)),
                    "r-",
                    label="Solved curve by 4 order R-K, a=-50"
                );
                plt.legend();
                plt.ylabel("log10(y)")
                plt.show();
            else:
                pass

        plot_se()
        plot_solvedValue()
        plot_solvedCurve()


case = RungeKutta()
case.fit()
case.plot_result()
\end{python}

import numpy as np


class ConditionNumber:
    """Applies an algorithm to calculate the condition number of any matrix.

    Attributes:
        __init__: A function initializing the class object
        read_data: A function loading matrix data
        get_condition_number: A function call to calculate condition number
        plot_result: A function displaying calculation results
    """

    def __init__(self, shape, generate_flag=True, hilbert_flag=False):
        """Initializes the class object by parameter assignment"""
        self.generate_flag = generate_flag
        self.Hilbert_flag = hilbert_flag
        self.shape = shape
        self.X = None
        self.conditionNumber = None

    def read_data(self):
        """Loads prepared data to fit"""
        if self.generate_flag:
            self.X = np.random.rand(*self.shape)
        elif self.Hilbert_flag:
            self.X = np.zeros(self.shape)
            for i in range(self.shape[0]):
                for j in range(self.shape[1]):
                    self.X[i, j] = 1 / (i + j + 1)
        else:
            pass

    def get_condition_number(self):
        """Calculates condition number"""
        self.conditionNumber = np.linalg.cond(self.X)

    def plot_result(self):
        """Plots specific results"""

        def plot_conditionNumbert():
            """Plots square error"""
            print("The matrix is: ")
            print(self.X)
            print("The shape of this matrix is {}".format(self.shape))
            print("The condition number of this matrix is {}".format(self.conditionNumber))

        plot_conditionNumbert()


np.set_printoptions(threshold=10)
case = ConditionNumber(shape=(27, 27))
case.read_data()
case.get_condition_number()
case.plot_result()
